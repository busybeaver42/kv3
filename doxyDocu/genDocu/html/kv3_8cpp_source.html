1259 inline k4a::Vector kv3::extract_gravity_from_imu(k4a::Vector& imuAcc)
1260 {
1261  //const k4a_calibration_t& sensorCalibration;
1262  const float* transformation_color_IMU_R = calibration.extrinsics[K4A_CALIBRATION_TYPE_ACCEL][K4A_CALIBRATION_TYPE_COLOR].rotation;
1263  k4a::Vector Rx = { transformation_color_IMU_R[0], transformation_color_IMU_R[1], transformation_color_IMU_R[2] };
1264  k4a::Vector Ry = { transformation_color_IMU_R[3], transformation_color_IMU_R[4], transformation_color_IMU_R[5] };
1265  k4a::Vector Rz = { transformation_color_IMU_R[6], transformation_color_IMU_R[7], transformation_color_IMU_R[8] };
1266 
1267  k4a::Vector depthAcc = { Rx.Dot(imuAcc), Ry.Dot(imuAcc) , Rz.Dot(imuAcc) };
1268  // The acceleration due to gravity, g, is in a direction toward the ground.
1269  // However an accelerometer at rest in a gravity field reports upward acceleration
1270  // relative to the local inertial frame (the frame of a freely falling object).
1271  k4a::Vector depthGravity = depthAcc.Normalized() * -1;
1272 
1273  return depthGravity;
1274 }
